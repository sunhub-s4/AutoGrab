-- =====================
-- GLOBAL EFFECT RUNNERS (one heartbeat per effect type)
-- =====================
local globalEffectRunners = {}

local function getOrStartEffectRunner(effectName, effectModule)
    if not globalEffectRunners[effectName] then
        local callbacks = {}
        local conn = RunService.Heartbeat:Connect(function(dt)
            local result = effectModule.simulate(dt)
            for _, cb in pairs(callbacks) do
                cb(result)
            end
        end)
        globalEffectRunners[effectName] = {
            conn = conn,
            callbacks = callbacks,
        }
    end
    return globalEffectRunners[effectName].callbacks
end

local function format(t)
    local m = math.floor(t / 60)
    local s = t % 60
    if m == 0 then return string.format("%02d", s) end
    return string.format("%02d:%02d", m, s)
end

local function slideIn(main)
    main.AnchorPoint = Vector2.new(0, 0)
    main.Position = UDim2.new(0, 0, 1.5, 0)
    TweenService:Create(main, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
        { Position = UDim2.new(0, 0, 0, 0) }):Play()
end

local function startTimer(frame, main, text, duration)
    if frame:GetAttribute("Running") then return end
    frame:SetAttribute("Running", true)
    local endTime = os.clock() + duration
    local connection
    connection = RunService.Heartbeat:Connect(function()
        local remaining = math.ceil(endTime - os.clock())
        if remaining <= 1 then
            text.Text = "01"
            frame:SetAttribute("Running", false)
            connection:Disconnect()
            for i, v in ipairs(activeUIs) do if v == frame then table.remove(activeUIs, i) break end end
            TweenService:Create(main, TweenInfo.new(0.4, Enum.EasingStyle.Quad, Enum.EasingDirection.In),
                { Position = UDim2.new(0, 0, 1.5, 0) }):Play()
            task.wait(0.4)
            frame.Visible = false
            main.Visible = false
            return
        end
        if math.floor(remaining / 60) == 0 then
            text.Text = string.format("%02d", remaining)
        else
            text.Text = format(remaining)
        end
    end)
end

local function getActiveEvents()
    local options = {}
    for _, child in ipairs(parent:GetChildren()) do
        if child:IsA("Frame") then table.insert(options, child.Name) end
    end
    return options
end

local function spawnEvent(eventName)
    local frame = parent:FindFirstChild(eventName)
    if not frame then return end
    local main = frame:WaitForChild("Main")
    local text = main:WaitForChild("TextLabel")
    if table.find(activeUIs, frame) then return end
    table.insert(activeUIs, frame)
    local index = #activeUIs
    frame.Visible = true
    main.Visible = true
    frame.ZIndex = 10 + index
    main.AnchorPoint = Vector2.new(0, 0)
    main.Position = UDim2.new(0, 0, 0, baseOffset + ((index - 1) * spacing))
    slideIn(main)
    startTimer(frame, main, text, timerDuration)
end

local function applyMutationToModel(model, animalName, mutationName)
    if not model or not mutationName then return end
    local trove = Trove.new()
    trove:Add(model.AncestryChanged:Connect(function(_, parent)
        if not parent then trove:Destroy() end
    end))

    local mutData = Mutations[mutationName]
    if not mutData then return trove end

    local mutSurface = MutSurfaces:FindFirstChild(animalName)
    local paletteAttr = model:GetAttribute(attr("*Palette", mutationName)) or model:GetAttribute("Palette") or 1
    local paletteIdx = tonumber(paletteAttr) or 1
    local safeIdx = (mutData.Palettes and mutData.Palettes[paletteIdx]) and paletteIdx or 1
    local skipped = {}

    local function ignoreColor(part)
        local a = part:GetAttribute(attr("*IgnoreColor", mutationName))
        if a == false then return false end
        return part:GetAttribute("IgnoreColor") or a
    end

    local function doPalettePass()
        local palette = mutData.Palettes[safeIdx]
        for _, part in model:GetDescendants() do
            if part:IsA("BasePart") and not ignoreColor(part) then
                local colorAttr = part:GetAttribute(attr("*Color", mutationName)) or part:GetAttribute("Color") or 1
                local colorIdx = tonumber(colorAttr) or 1
                local color = palette[colorIdx] or palette[math.clamp(colorIdx, 1, #palette)]

                if part:GetAttribute("Neon") then
                    local oldMat = part.Material
                    part.Material = Enum.Material.Neon
                    trove:Add(function() part.Material = oldMat end)
                end

                local sa = part:FindFirstChildOfClass("SurfaceAppearance")
                if sa then
                    sa:Destroy()
                    if mutSurface then
                        local newSa = trove:Clone(mutSurface)
                        newSa.Color = (mutationName == "Divine") and palette[1] or color
                        newSa.Parent = part
                    end
                    trove:Add(function()
                        local existing = part:FindFirstChildOfClass("SurfaceAppearance")
                        if existing then existing:Destroy() end
                        sa:Clone().Parent = part
                    end)
                else
                    local oldVariant = part.MaterialVariant
                    if oldVariant == "Strawberry Stud Light" or oldVariant == "Strawberry Stud Dark" then
                        skipped[part] = true
                        part.MaterialVariant = attr("* Strawberry Stud Light", mutationName)
                        trove:Add(function() part.MaterialVariant = oldVariant end)
                    else
                        local oldColor = part.Color
                        part.Color = color
                        trove:Add(function() part.Color = oldColor end)
                    end
                end
            end
        end
    end

    local function doVfxAndTag()
        local vfxFolder = Vfx:FindFirstChild(mutationName)
        local vfxInstance = model:FindFirstChild("VfxInstance")
        if vfxFolder and vfxInstance then
            for _, vfx in vfxFolder:GetChildren() do
                trove:Clone(vfx).Parent = vfxInstance
            end
        end
        model:SetAttribute("__mutation", mutationName)
        trove:Add(function() model:SetAttribute("__mutation", nil) end)
    end

    if mutationName == "Rainbow" then
        trove:Add(function() model:RemoveTag("RainbowModel") end)
        for _, part in model:GetDescendants() do
            if part:IsA("BasePart") and not part:GetAttribute("IgnoreRainbowColor") then
                local sa = part:FindFirstChildOfClass("SurfaceAppearance")
                if sa then
                    trove:Add(function()
                        local existing = part:FindFirstChildOfClass("SurfaceAppearance")
                        if existing then existing:Destroy() end
                        sa:Clone().Parent = part
                    end)
                else
                    local oldColor = part.Color
                    trove:Add(function() part.Color = oldColor end)
                    local oldVariant = part.MaterialVariant
                    if oldVariant == "Strawberry Stud Light" or oldVariant == "Strawberry Stud Dark" then
                        part.MaterialVariant = ""
                        trove:Add(function() part.MaterialVariant = oldVariant end)
                    end
                end
            end
        end
        model:AddTag("RainbowModel")
        local heartbeat
        heartbeat = RunService.Heartbeat:Connect(function()
            local hue = tick() % 5 / 5
            for _, tagged in CollectionService:GetTagged("RainbowModel") do
                if tagged == model then
                    for _, part in tagged:GetDescendants() do
                        if part:IsA("BasePart") and not part:GetAttribute("IgnoreRainbowColor") and not part:FindFirstChildOfClass("SurfaceAppearance") then
                            part.Color = Color3.fromHSV(hue, 1, 1)
                        end
                    end
                end
            end
        end)
        trove:Add(heartbeat)
        return trove
    end

    -- Palette pass must run first so skipped[] is populated before
    -- the mutation-specific material overrides below check it.
    doPalettePass()

    if mutationName == "Lava" then
        for _, part in model:GetDescendants() do
            if part:IsA("BasePart") and not skipped[part] and not ignoreColor(part) then
                if (part:GetAttribute(attr("*Color", mutationName)) or part:GetAttribute("Color")) == 1 then
                    local oldMat = part.Material
                    part.Material = Enum.Material.Neon
                    trove:Add(function() part.Material = oldMat end)
                end
            end
        end
    elseif mutationName == "Galaxy" then
        for _, part in model:GetDescendants() do
            if part:IsA("BasePart") and not skipped[part] and not ignoreColor(part) then
                if (part:GetAttribute(attr("*Color", mutationName)) or part:GetAttribute("Color")) == 1 then
                    local oldMat = part.Material
                    part.Material = Enum.Material.Neon
                    trove:Add(function() part.Material = oldMat end)
                end
                local oldVariant = part.MaterialVariant
                part.MaterialVariant = "Galaxy Stud"
                trove:Add(function() part.MaterialVariant = oldVariant end)
            end
        end
    elseif mutationName == "YinYang" then
        for _, part in model:GetDescendants() do
            if part:IsA("BasePart") and not skipped[part] and not ignoreColor(part) then
                local colorVal = part:GetAttribute(attr("*Color", mutationName)) or part:GetAttribute("Color") or 1
                if colorVal == 3 or colorVal == 4 then
                    local oldMat = part.Material
                    part.Material = Enum.Material.Neon
                    trove:Add(function() part.Material = oldMat end)
                end
            end
        end
    elseif mutationName == "Radioactive" then
        for _, part in model:GetDescendants() do
            if part:IsA("BasePart") and not skipped[part] and not ignoreColor(part) then
                local oldVariant = part.MaterialVariant
                local oldMat = part.Material
                local colorVal = part:GetAttribute(attr("*Color", mutationName)) or part:GetAttribute("Color") or 1
                local neon = (colorVal == 2 and (safeIdx == 1 or safeIdx == 4)) or (colorVal == 2 and safeIdx == 5) or (colorVal == 6 and safeIdx == 5)
                if neon then
                    part.Material = Enum.Material.Neon
                    trove:Add(function() part.Material = oldMat end)
                end
                local studAttr = part:GetAttribute(attr("*Stud", mutationName))
                if studAttr == false or (part.MaterialVariant ~= "Custom Stud" and studAttr ~= true) or part:GetAttribute(attr("*Ignore", mutationName)) then
                    if studAttr == false then
                        part.MaterialVariant = ""
                        trove:Add(function() part.MaterialVariant = oldVariant end)
                    end
                elseif part:GetAttribute(attr("*MaterialMode", mutationName)) == 2 or model:GetAttribute(attr("*MaterialMode", mutationName)) == 2 then
                    local studName = (part:GetAttribute(attr("*Material", mutationName)) == 2 or model:GetAttribute(attr("*Material", mutationName)) == 2) and "Radioactive Stud" or "Radioactive Stud2"
                    part.Material = Enum.Material.SmoothPlastic
                    part.MaterialVariant = studName
                    trove:Add(function() part.MaterialVariant = oldVariant; part.Material = oldMat end)
                elseif colorVal ~= 2 and colorVal ~= 6 then
                    local studName = (part:GetAttribute(attr("*Material", mutationName)) == 2 or model:GetAttribute(attr("*Material", mutationName)) == 2) and "Radioactive Stud" or "Radioactive Stud2"
                    part.Material = Enum.Material.SmoothPlastic
                    part.MaterialVariant = studName
                    trove:Add(function() part.MaterialVariant = oldVariant; part.Material = oldMat end)
                end
            end
        end
    elseif mutationName == "Cursed" then
        local zombieTrait = model:FindFirstChild("_Trait.Zombie")
        if zombieTrait then zombieTrait:Destroy() end
        for _, part in model:GetDescendants() do
            if part:IsA("BasePart") and not skipped[part] and not ignoreColor(part) then
                local oldVariant = part.MaterialVariant
                local oldMat = part.Material
                local oldColor = part.Color
                local colorVal = part:GetAttribute(attr("*Color", mutationName)) or part:GetAttribute("Color") or 1
                local neon = (colorVal == 2 and (safeIdx == 1 or safeIdx == 4)) or (colorVal == 2 and safeIdx == 5) or (colorVal == 6 and safeIdx == 5)
                if neon then
                    part.Material = Enum.Material.Neon
                    trove:Add(function() part.Material = oldMat end)
                end
                local studAttr = part:GetAttribute(attr("*Stud", mutationName))
                if studAttr == false or (part.MaterialVariant ~= "Custom Stud" and studAttr ~= true) or part:GetAttribute(attr("*Ignore", mutationName)) then
                    if studAttr == false then
                        part.MaterialVariant = ""
                        trove:Add(function() part.MaterialVariant = oldVariant end)
                    end
                elseif colorVal ~= 2 and colorVal ~= 6 or studAttr == true then
                    part.Material = Enum.Material.SmoothPlastic
                    part.MaterialVariant = "Cursed Stud"
                    part.Color = Color3.fromRGB(255, 23, 23)
                    trove:Add(function() part.MaterialVariant = oldVariant; part.Material = oldMat; part.Color = oldColor end)
                end
                local sa = part:FindFirstChildOfClass("SurfaceAppearance")
                if sa then
                    if not part:GetAttribute(attr("*IgnoreSurfaceColor", mutationName)) then
                        sa.Color = Color3.fromRGB(255, 23, 23)
                    end
                    if part:GetAttribute("IgnoreSurface") then
                        sa:Destroy()
                    end
                end
            end
        end
        local zombieFolder = ReplicatedStorage.Models.TraitsPerAnimal.Zombie:FindFirstChild(animalName)
        if zombieFolder then
            local zombieClone = trove:Clone(zombieFolder)
            zombieClone.Name = "_Trait.Zombie"
            zombieClone.Parent = model
            local attached = {}
            for _, piece in zombieClone:GetChildren() do
                piece.Color = Color3.fromRGB(255, 23, 23)
                local att = piece:FindFirstChildOfClass("Attachment")
                if att then
                    local target = attached[att.Name] or model:FindFirstChild(att.Name, true)
                    if target then
                        piece:PivotTo(target.WorldCFrame)
                        attached[att.Name] = target
                        local rc = Instance.new("RigidConstraint")
                        rc.Attachment0 = att
                        rc.Attachment1 = target
                        rc.Parent = att
                    else
                        piece:Destroy()
                    end
                end
            end
        end
    elseif mutationName == "Divine" then
        local emissiveStrength = model:GetAttribute("EmissiveStrength") or 2
        for _, desc in model:GetDescendants() do
            if desc:IsA("SurfaceAppearance") then
                local oldEmissive = desc.EmissiveStrength
                desc.EmissiveStrength = emissiveStrength
                trove:Add(function() desc.EmissiveStrength = oldEmissive end)
            end
            if desc:IsA("BasePart") and not skipped[desc] and not ignoreColor(desc) then
                local oldMat = desc.Material
                local oldVariant = desc.MaterialVariant
                local colorVal = desc:GetAttribute(attr("*Color", mutationName)) or desc:GetAttribute("Color") or 1
                local neon = (colorVal == 2 and (safeIdx == 1 or safeIdx == 4)) or (colorVal == 2 and safeIdx == 5) or (colorVal == 6 and safeIdx == 5)
                if neon then
                    desc.Material = Enum.Material.Neon
                    trove:Add(function() desc.Material = oldMat end)
                end
                local studAttr = desc:GetAttribute(attr("*Stud", mutationName))
                if studAttr == false or (desc.MaterialVariant ~= "Custom Stud" and studAttr ~= true) or desc:GetAttribute(attr("*Ignore", mutationName)) then
                    if studAttr == false then
                        desc.MaterialVariant = ""
                        trove:Add(function() desc.MaterialVariant = oldVariant end)
                    end
                elseif desc:GetAttribute(attr("*MaterialMode", mutationName)) == 2 or model:GetAttribute(attr("*MaterialMode", mutationName)) == 2 then
                    desc.Material = Enum.Material.SmoothPlastic
                    desc.MaterialVariant = "Divine Stud"
                    trove:Add(function() desc.MaterialVariant = oldVariant; desc.Material = oldMat end)
                elseif colorVal ~= 2 and colorVal ~= 6 then
                    desc.Material = Enum.Material.SmoothPlastic
                    desc.MaterialVariant = "Divine Stud"
                    trove:Add(function() desc.MaterialVariant = oldVariant; desc.Material = oldMat end)
                end
            end
        end
    end

    if mutationName ~= "Rainbow" then
        doVfxAndTag()
    end

    return trove
end

-- ============================================================
local function applyTraitsToModel(model, animalName, traitsList)
    if not model or #traitsList == 0 then return end
    local trove = Trove.new()
    trove:Add(model.AncestryChanged:Connect(function(_, parent)
        if not parent then trove:Destroy() end
    end))

    local traitsSet = {}
    for _, t in traitsList do traitsSet[t] = true end

    local skibidiModel = traitsSet.Skibidi and ReplicatedStorage.Models.TraitsPerAnimal.Skibidi:FindFirstChild(animalName)

    for _, traitName in traitsList do
        local traitData = Traits[traitName]
        if not traitData then continue end

        local perAnimalFolder = TraitsPerAnimal:FindFirstChild(traitName)
        if perAnimalFolder then
            local perAnimalModel = perAnimalFolder:FindFirstChild(animalName)
            if perAnimalModel and not model:FindFirstChild(("_Trait.%s"):format(traitName)) then
                local clone = trove:Clone(perAnimalModel)
                clone.Name = ("_Trait.%s"):format(traitName)
                for _, child in clone:GetChildren() do
                    local att = child:FindFirstChildOfClass("Attachment")
                    if att then
                        local target = model:FindFirstChild(att.Name, true)
                        if target then
                            local rc = Instance.new("RigidConstraint")
                            rc.Attachment0 = att
                            rc.Attachment1 = target
                            rc.Parent = child
                        end
                    end
                end
                clone.Parent = model

local function attachOverheadToHat(model, animalName, priceValue, mutationName, traitsList, rarityName)
    local hatAttachment = nil
    local function searchForHatAttachment(instance)
        for _, child in ipairs(instance:GetChildren()) do
            if child:IsA("Attachment") and child.Name == "HatAttachment" then
                hatAttachment = child
                return true
            end
            if searchForHatAttachment(child) then return true end
        end
        return false
    end
    searchForHatAttachment(model)
    if not hatAttachment then
        warn("No HatAttachment found in", model.Name)
        return nil
    end

    local controller = ReplicatedStorage:WaitForChild("Controllers")
    local fastController = controller:WaitForChild("FastOverheadController")
    local partTemplate = fastController:WaitForChild("FastOverheadTemplate")
    local guiTemplate = fastController:WaitForChild("AnimalOverhead")

    local part = partTemplate:Clone()
    part.Anchored = true
    part.CanCollide = false
    part.Transparency = 1
    part.Parent = workspace

    local surfaceGui = guiTemplate:Clone()
    surfaceGui.Parent = part
    surfaceGui.Adornee = part
    surfaceGui.Face = Enum.NormalId.Front
    surfaceGui.AlwaysOnTop = false

    local displayName = surfaceGui:WaitForChild("DisplayName")
    displayName.Text = animalName

    -- Generation
    local generation = surfaceGui:WaitForChild("Generation")
    local animalDataModule = require(ReplicatedStorage:WaitForChild("Datas"):WaitForChild("Animals"))
    local animalData = animalDataModule[animalName]
    local generationValue = animalData and animalData.Generation or 0

    local function formatGen(n)
        n = tonumber(n) or 0
        if n >= 1e12 then return string.format("$%gT/s", math.floor(n / 1e12))
        elseif n >= 1e9 then return string.format("$%gB/s", math.floor(n / 1e9))
        elseif n >= 1e6 then return string.format("$%gM/s", math.floor(n / 1e6))
        elseif n >= 1e3 then return string.format("$%gK/s", math.floor(n / 1e3))
        else return "$" .. n .. "/s" end
    end

    local mutModifier = mutationName and MutationDisplayData[mutationName] and MutationDisplayData[mutationName].Modifier or 0
    local traitModifier = 0
    if traitsList then
        for _, traitName in ipairs(traitsList) do
            local td = TraitDisplayData[traitName]
            if td and td.MultiplierModifier then
                traitModifier = traitModifier + td.MultiplierModifier
            end
        end
    end
    local totalModifier = 1 + mutModifier + traitModifier
    generation.Text = formatGen(generationValue * totalModifier)

    -- Price
    local price = surfaceGui:WaitForChild("Price")
    local function formatPrice(n)
        n = tonumber(n) or 0
        if n >= 1e9 then return string.format("$%dB", n / 1e9)
        elseif n >= 1e6 then return string.format("$%dM", n / 1e6)
        elseif n >= 1e3 then return string.format("$%dK", n / 1e3)
        else return "$" .. n end
    end
    price.Text = formatPrice(priceValue)

    -- Rarity
    local rarity = surfaceGui:WaitForChild("Rarity")
    rarity.Text = rarityName

    local EFFECT_MAP = {
        ["OG"]     = "OG",
        ["Secret"] = "Zebra",
    }

    local effectName = EFFECT_MAP[rarityName]
    local cleanupFn = nil
    local applyResult = nil
    local effectModule = nil

    if effectName then
        local Effects = ReplicatedStorage:WaitForChild("Packages"):WaitForChild("Gradients"):WaitForChild("Effects")
        local effectFolder = Effects:FindFirstChild(effectName)
        if effectFolder then
            effectModule = require(effectFolder)
            applyResult = effectModule.apply(rarity)
            cleanupFn = applyResult.cleanup

            local callbacks = getOrStartEffectRunner(effectName, effectModule)
            local cbKey = tostring(part)
            callbacks[cbKey] = function(result)
                if not rarity or not rarity.Parent then
                    callbacks[cbKey] = nil
                    return
                end
                if applyResult.main and result.main then
                    applyResult.main.Color = result.main
                end
                if applyResult.stroke and result.stroke then
                    applyResult.stroke.Color = result.stroke
                end
            end

            model.AncestryChanged:Connect(function(_, parent)
                if not parent then
                    callbacks[cbKey] = nil
                end
            end)

            part.Destroying:Connect(function()
                callbacks[cbKey] = nil
            end)
        end
    end

    -- Mutation label
    local mutationLabel = surfaceGui:FindFirstChild("Mutation")
    if mutationLabel and mutationLabel:IsA("TextLabel") then
        local mutData = MutationDisplayData[mutationName]
        if mutData and mutData.DisplayWithRichText then
            mutationLabel.RichText = true
            mutationLabel.Text = mutData.DisplayWithRichText
            mutationLabel.Visible = true
        elseif mutationName and mutationName ~= "None" then
            mutationLabel.RichText = false
            mutationLabel.Text = mutationName
            mutationLabel.Visible = true
        else
            mutationLabel.Visible = false
        end
    end

    -- Traits frame
    local traitsFrame = surfaceGui:FindFirstChild("Traits")
    if traitsFrame and traitsFrame:IsA("Frame") then
        if traitsList and #traitsList > 0 then
            traitsFrame.Visible = true
            for _, child in ipairs(traitsFrame:GetChildren()) do
                if child:IsA("ImageLabel") and child.Name ~= "Template" then
                    child:Destroy()
                end
            end
            local template = traitsFrame:FindFirstChild("Template")
            if template and template:IsA("ImageLabel") then
                local layout = traitsFrame:FindFirstChildOfClass("UIListLayout")
                if not layout then
                    layout = Instance.new("UIListLayout")
                    layout.FillDirection = Enum.FillDirection.Horizontal
                    layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
                    layout.VerticalAlignment = Enum.VerticalAlignment.Center
                    layout.Padding = UDim.new(0, 5)
                    layout.Parent = traitsFrame
                end
                for _, traitName in ipairs(traitsList) do
                    local icon = template:Clone()
                    icon.Name = traitName
                    icon.Visible = true
                    local td = TraitDisplayData[traitName]
                    if td and td.Icon then
                        icon.Image = td.Icon
                    else
                        icon.Image = "rbxassetid://0"
                    end
                    icon.Parent = traitsFrame
                end
            end
        else
            traitsFrame.Visible = false
        end
    else
        warn("Traits frame not found in AnimalOverhead")
    end

    -- Follow attachment
    local connection
    connection = RunService.Stepped:Connect(function()
        if not hatAttachment or not hatAttachment.Parent or not model.Parent then
            if connection then connection:Disconnect() end
            if cleanupFn then cleanupFn() end
            part:Destroy()
            surfaceGui:Destroy()
            return
        end

        local camera = workspace.CurrentCamera
        if not camera then return end

        local attachPos = hatAttachment.WorldPosition
        local partHeight = part.Size.Y
        local offset = Vector3.new(0, partHeight / 2 + 0.3, 0)
        local position = attachPos + offset

        part.CFrame = CFrame.lookAlong(position, -camera.CFrame.LookVector)
    end)

    model.AncestryChanged:Connect(function(_, parent)
        if not parent then
            if connection then connection:Disconnect() end
            if cleanupFn then cleanupFn() end
            part:Destroy()
            surfaceGui:Destroy()
        end
    end)

    part.Destroying:Connect(function()
        if cleanupFn then cleanupFn() end
    end)

    return part
end

BrickColor.new("Bright violet").Color,

BrickColor.new("Brick yellow").Color,

local function GetNameValue(pName)
local value = 0
for i = 1, #pName do
local cValue = string.byte(pName:sub(i, i))
local reverseIndex = #pName - i + 1
if #pName % 2 == 1 then reverseIndex = reverseIndex - 1 end
if reverseIndex % 4 >= 2 then
cValue = -cValue
end
value = value + cValue
end
return value
end

local function ComputeNameColor(pName)
return NAME_COLORS[(GetNameValue(pName) % #NAME_COLORS) + 1]
end

local function ColorToHex(color)
return string.format("#%02X%02X%02X",
math.floor(color.R * 255),
math.floor(color.G * 255),
math.floor(color.B * 255)
)
end

local function GetClosestPlayer()
local char = player.Character
if not char then return end

local root = char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Head")
if not root then return end

local closest, closestDist = nil, math.huge

for _, plr in ipairs(Players:GetPlayers()) do
if plr ~= player then
local otherChar = plr.Character
if otherChar and otherChar:FindFirstChild("HumanoidRootPart") then
local dist = (root.Position - otherChar.HumanoidRootPart.Position).Magnitude
if dist < closestDist then
closest = plr
closestDist = dist
end
end
end
end

return closest, closestDist
end
