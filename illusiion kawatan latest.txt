local function loadConfig()
    pcall(function()
        if not (isfile and isfile(ConfigFileName)) then return end
        local raw  = readfile(ConfigFileName)
        local data = HttpService:JSONDecode(raw)
        if data then
            for k,v in pairs(data.Config   or {}) do if Config[k]~=nil   then Config[k]=v   end end
            for k,v in pairs(data.Keybinds or {}) do if Keybinds[k]~=nil then Keybinds[k]=v end end
            if data.Positions then GuiPositions=data.Positions end
            if data.AD_WP then
                for dir,wps in pairs(data.AD_WP) do
                    if AD_WP[dir] then
                        for i,wp in pairs(wps) do
                            if AD_WP[dir][i] then
                                if wp.x then AD_WP[dir][i].x=wp.x end
                                if wp.z then AD_WP[dir][i].z=wp.z end
                            end
                        end
                    end
                end
            end
        end
    end)
end

local function isMyPlotByName(pn)
    local ct = tick()
    if Steal.plotCache[pn] and (ct-(Steal.plotCacheTime[pn] or 0))<PLOT_CACHE_DURATION then 
        return Steal.plotCache[pn] 
    end
    local plots = Workspace:FindFirstChild("Plots")
    if not plots then Steal.plotCache[pn]=false; Steal.plotCacheTime[pn]=ct; return false end
    local plot = plots:FindFirstChild(pn)
    if not plot then Steal.plotCache[pn]=false; Steal.plotCacheTime[pn]=ct; return false end
    local sign = plot:FindFirstChild("PlotSign")
    if sign then
        local yb = sign:FindFirstChild("YourBase")
        if yb and yb:IsA("BillboardGui") then
            local r = yb.Enabled==true
            Steal.plotCache[pn]=r
            Steal.plotCacheTime[pn]=ct
            return r
        end
    end
    Steal.plotCache[pn]=false
    Steal.plotCacheTime[pn]=ct
    return false
end

local function findNearestPrompt()
    local char = lp.Character; if not char then return nil end
    local root = char:FindFirstChild("HumanoidRootPart"); if not root then return nil end
    local ct = tick()
    
    if ct-Steal.promptCacheTime<PROMPT_CACHE_REFRESH and #Steal.cachedPrompts>0 then
        local np,nd = nil,math.huge
        for _,data in ipairs(Steal.cachedPrompts) do
            if data.spawn then
                local dist = (data.spawn.Position-root.Position).Magnitude
                if dist<=Steal.StealRadius and dist<nd then 
                    np=data.prompt
                    nd=dist 
                end
            end
        end
        if np then return np end
    end
    
    Steal.cachedPrompts={}
    Steal.promptCacheTime=ct
    local plots = Workspace:FindFirstChild("Plots"); if not plots then return nil end
    local np,nd = nil,math.huge
    
    for _,plot in ipairs(plots:GetChildren()) do
        if isMyPlotByName(plot.Name) then continue end
        local pods = plot:FindFirstChild("AnimalPodiums"); if not pods then continue end
        for _,pod in ipairs(pods:GetChildren()) do
            pcall(function()
                local base = pod:FindFirstChild("Base")
                local sp = base and base:FindFirstChild("Spawn")
                if sp then
                    local att = sp:FindFirstChild("PromptAttachment")
                    if att then
                        for _,child in ipairs(att:GetChildren()) do
                            if child:IsA("ProximityPrompt") then
                                local dist = (sp.Position-root.Position).Magnitude

                                    np=child

                                end

                            end

                    end

local function executeSteal(prompt)
    local ct = tick()
    if ct-lastStealTick<STEAL_COOLDOWN then return end
    if isStealing then return end
    
    if not Steal.Data[prompt] then
        Steal.Data[prompt]={hold={},trigger={},ready=true}
        pcall(function()
            if getconnections then
                for _,c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do 
                    if c.Function then table.insert(Steal.Data[prompt].hold,c.Function) end 
                end
                for _,c in ipairs(getconnections(prompt.Triggered)) do 
                    if c.Function then table.insert(Steal.Data[prompt].trigger,c.Function) end 
                end
            else 
                Steal.Data[prompt].useFallback=true 
            end
        end)
    end
    
    local data = Steal.Data[prompt]
    if not data.ready then return end
    
    data.ready=false
    isStealing=true
    stealStartTime=ct
    lastStealTick=ct
    startStealBarProgress(Steal.StealDuration)
    
    task.spawn(function()
        local ok = false
        pcall(function()
            if not data.useFallback then
                for _,fn in ipairs(data.hold) do task.spawn(fn) end
                task.wait(Steal.StealDuration)
                for _,fn in ipairs(data.trigger) do task.spawn(fn) end
                ok=true
            end
        end)
        
        if not ok and fireproximityprompt then 
            pcall(function() 
                fireproximityprompt(prompt)
                ok=true 
            end) 
        end
        
        if not ok then 
            pcall(function() 
                prompt:InputHoldBegin()
                task.wait(Steal.StealDuration)
                prompt:InputHoldEnd() 
            end) 
        end
        
        task.wait(Steal.StealDuration*0.3)
        if stealBarConn then stealBarConn:Disconnect() end
        if GrabFill then GrabFill.Size=UDim2.new(0,0,1,0) end
        task.wait(0.05)
        data.ready=true
        isStealing=false
    end)
end

        if not isMyPlotByName(plot.Name) then
            local pods=plot:FindFirstChild("AnimalPodiums"); if not pods then continue end
            for _,pod in ipairs(pods:GetChildren()) do
                local base=pod:FindFirstChild("Base"); if not base then continue end
                local spawn=base:FindFirstChild("Spawn"); if not spawn then continue end
                if (spawn.Position-hrp.Position).Magnitude<=Steal.StealRadius then 
                    return true 
                end
            end
        end
    end
    return false
end

local medusaCounterEnabled = false
local medusaDebounce = false
local medusaLastUsed = 0
local medusaConns = {}
local MEDUSA_COOLDOWN = 25

local function findMedusa()
    local char = lp.Character; if not char then return nil end
    for _,t in ipairs(char:GetChildren()) do 
        if t:IsA("Tool") and t.Name:lower():find("medusa") then 
            return t 
        end 
    end
    local bp = lp:FindFirstChild("Backpack")
    if bp then 
        for _,t in ipairs(bp:GetChildren()) do 
            if t:IsA("Tool") and t.Name:lower():find("medusa") then 
                return t 
            end 
        end 
    end
    return nil
end

local function useMedusa()
    if medusaDebounce or tick()-medusaLastUsed<MEDUSA_COOLDOWN then return end
    local char = lp.Character; if not char then return end
    medusaDebounce = true
    local med = findMedusa()
    if med then
        if med.Parent ~= char then 
            local h = char:FindFirstChildOfClass("Humanoid")
            if h then h:EquipTool(med) end 
        end
        pcall(function() med:Activate() end)
        medusaLastUsed = tick()
    end
    medusaDebounce = false
end

local function onAnchorChanged(part)
    return part:GetPropertyChangedSignal("Anchored"):Connect(function()
        if part.Anchored and part.Transparency==1 and medusaCounterEnabled then 
            useMedusa() 
        end
    end)
end

local function setupMedusa(char)
    for _,c in pairs(medusaConns) do 
        pcall(function() c:Disconnect() end) 
    end
    medusaConns = {}
    if not char then return end
    
    for _,part in ipairs(char:GetDescendants()) do 
        if part:IsA("BasePart") then 
            table.insert(medusaConns, onAnchorChanged(part)) 
        end 
    end
    
    table.insert(medusaConns, char.DescendantAdded:Connect(function(part)
        if part:IsA("BasePart") then 
            table.insert(medusaConns, onAnchorChanged(part)) 
        end
    end))
end

local function enableMedusaCounter()
    medusaCounterEnabled = true
    Config.MEDUSA_COUNTER_ENABLED = true
    setupMedusa(lp.Character)
end

local function disableMedusaCounter()
    medusaCounterEnabled = false
    Config.MEDUSA_COUNTER_ENABLED = false
    for _,c in pairs(medusaConns) do 
        pcall(function() c:Disconnect() end) 
    end
    medusaConns = {}
end

local function getNearestPlayerForBat()
    local character=lp.Character; if not character then return nil end
    local hrp=character:FindFirstChild("HumanoidRootPart"); if not hrp then return nil end
    local nearestPlayer,nearestDistance=nil,math.huge
    local myPos=hrp.Position
    for _,p in pairs(Players:GetPlayers()) do
        if p~=lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local pHum=p.Character:FindFirstChildOfClass("Humanoid")
            if pHum and pHum.Health>0 then
                local d=(p.Character.HumanoidRootPart.Position-myPos).Magnitude
                if d<nearestDistance then nearestDistance=d; nearestPlayer=p end
            end
        end
    end
    return nearestPlayer
end

local fpsSamples=0; local fpsCount=0; local pingTimer=0; local barTime=0
local function getNearestStealDist()
    local char=lp.Character; if not char then return math.huge end
    local myhrp=char:FindFirstChild("HumanoidRootPart"); if not myhrp then return math.huge end
    local plots=Workspace:FindFirstChild("Plots"); if not plots then return math.huge end
    local nearest=math.huge
    for _,plot in ipairs(plots:GetChildren()) do
        if not isMyPlotByName(plot.Name) then
            local podiums=plot:FindFirstChild("AnimalPodiums"); if not podiums then continue end
            for _,pod in ipairs(podiums:GetChildren()) do
                pcall(function() local base=pod:FindFirstChild("Base"); local spn=base and base:FindFirstChild("Spawn"); if spn then local d=(spn.Position-myhrp.Position).Magnitude; if d<nearest then nearest=d end end end)
            end
        end
    end; return nearest
end

RunService.Heartbeat:Connect(function(dt)
    fpsSamples=fpsSamples+(1/math.max(dt,0.001)); fpsCount=fpsCount+1
    if fpsCount>=10 then FPSLabel.Text="FPS "..math.floor(fpsSamples/fpsCount); fpsSamples=0; fpsCount=0 end
    pingTimer=pingTimer+dt
    if pingTimer>=1 then pingTimer=0; pcall(function() local pi=Stats.Network.ServerStatsItem["Data Ping"]; if pi then PingLabel.Text="PING "..math.floor(pi:GetValue()).."ms" end end) end
    if not stealBarActive then
        if Steal.AutoStealEnabled then
            local radius=Steal.StealRadius; local dist=getNearestStealDist()
            if dist<=radius then
                local closeness=1-math.clamp(dist/radius,0,1); barTime=barTime+dt*(1+closeness*4)
                local wave=math.abs(math.sin(barTime*math.pi)); local fill=wave*(0.15+closeness*0.85)
                GrabFill.Size=UDim2.new(fill,0,1,0); GrabFill.BackgroundColor3=Color3.fromRGB(math.floor(255-closeness*155),255,math.floor(255-closeness*155))
            else barTime=barTime+dt*0.8; local pulse=math.abs(math.sin(barTime*1.2))*0.08; GrabFill.Size=UDim2.new(pulse,0,1,0); GrabFill.BackgroundColor3=COLOR_WHITE end
        else GrabFill.Size=UDim2.new(0,0,1,0) end
    end
    if Workspace.CurrentCamera then Workspace.CurrentCamera.FieldOfView=Config.FOV_PLAYER end
    local hrp=getHRP()
    if hrp and speedBillboardLabel then
        local vel=hrp.AssemblyLinearVelocity
        local flatSpeed=Vector3.new(vel.X,0,vel.Z).Magnitude
        speedBillboardLabel.Text=string.format("%.1f sp",flatSpeed)
    end
    attachBillboards()
end)

UserInputService.InputBegan:Connect(function(input,gpe)
    if gpe then return end; if input.KeyCode==Enum.KeyCode.Unknown then return end
    local keyName=input.KeyCode.Name
    if keyName==Keybinds.UIToggle then MainMenu.Visible=not MainMenu.Visible
    elseif keyName==Keybinds.Speed then Config.SPEED_CUSTOMIZER_ENABLED=not Config.SPEED_CUSTOMIZER_ENABLED; speedFrame.Visible=Config.SPEED_CUSTOMIZER_ENABLED; if Config.SPEED_CUSTOMIZER_ENABLED then enableSpeedCustomizer(); spCollapsed=false; applySpeedCollapse() else disableSpeedCustomizer() end; updateSpToggle()
    elseif keyName==Keybinds.InfJump then if infJumpEnabled then disableInfJump() else enableInfJump() end
    elseif keyName==Keybinds.TPDown and Config.TP_DOWN_ENABLED then runTPDown()
    elseif keyName==Keybinds.DropBrainrot and Config.DROP_BRAINROT_ENABLED then runDropBrainrot()
    elseif keyName==Keybinds.BatAimbot then toggleBatAimbot()
    elseif keyName==Keybinds.Desync then Config.DESYNC_ENABLED=not Config.DESYNC_ENABLED; if Config.DESYNC_ENABLED then enableDesync() else disableDesync() end; updateDesyncBtn()
    elseif keyName==Keybinds.AutoDuel then
        if not AutoDuel or not AutoDuel.Enabled then return end
        AutoDuel.Active=not AutoDuel.Active; AutoDuel.Phase=1
        if AutoDuel.Active then
            if BatAimbot and BatAimbot.Active then stopBatAimbot() end
            setAutoDuelMovement(true)
        else
            local char=lp.Character; if char then local hrp=char:FindFirstChild("HumanoidRootPart"); if hrp then hrp.AssemblyLinearVelocity=Vector3.zero end end
            setAutoDuelMovement(false)
        end
        updateADToggle()
    elseif keyName==Keybinds.SpeedLagger then
        toggleSpeedLagger()
    end
end)

local function setupCharacter(char)
    task.wait(0.5)
    if AutoDuel.Active then
        AutoDuel.Active=false; AutoDuel.Phase=1
        if AutoDuel.lockConn then AutoDuel.lockConn:Disconnect(); AutoDuel.lockConn=nil end
        updateADToggle()
    end
    if BatAimbot.Active then stopBatAimbot() end
    if Config.NO_WALK_ANIMATION_ENABLED  then startUnwalk()            end
    if medusaCounterEnabled              then setupMedusa(char)        end
    if Config.HARDER_HIT_ANIM_ENABLED    then applyAnimPack(char)      end
    if Config.REMOVE_ACCESSORIES_ENABLED then removeAccessories()       end
    if Config.HIT_CIRCLE_ENABLED         then enableHitCircle()        end
    if Config.NO_CAM_COLLISION_ENABLED   then disableNoCameraCollision(); task.wait(0.1); enableNoCameraCollision() end
    if Config.SPEED_CUSTOMIZER_ENABLED   then enableSpeedCustomizer()  end
    if Config.SPEED_LAGGER_ENABLED       then speedLaggerEnabled=true  end
    dropBrainrotActive=false; attachBillboards()
end
